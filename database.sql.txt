
CREATE TABLE Categorie (
    id_categoria SERIAL PRIMARY KEY,
    nome VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE Spese (
    id_spesa SERIAL PRIMARY KEY,
    data DATE NOT NULL,
    importo DECIMAL(10, 2) NOT NULL CHECK (importo > 0),
    id_categoria INTEGER REFERENCES Categorie(id_categoria),
    descrizione TEXT
);

CREATE TABLE budget (
    id_budget SERIAL PRIMARY KEY,
    id_categoria INTEGER REFERENCES Categorie(id_categoria),
    importo_limite DECIMAL(10,2) NOT NULL,
    anno_mese varchar(7) NOT NULL, 
    CONSTRAINT unique_budget_periodo UNIQUE(id_categoria, anno_mese)
);
